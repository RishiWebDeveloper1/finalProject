<body>
    <!-- Global Search Component HTML -->
    <div class="global-search-container">
        <form class="global-search-form" id="globalSearchForm">
            <input type="text" id="globalSearchInput" class="global-search-input" placeholder="Search products..." autocomplete="off">
            <button type="submit" id="globalSearchButton" class="global-search-button">Search</button>
        </form>
        <div class="global-search-results" id="globalSearchResults"></div>
    </div>

    <script>
        /**
         * Global Search Component
         * 
         * This script creates a reusable search component that integrates with
         * your existing product database and can be used across all pages.
         */
        class GlobalSearch {
            constructor(options = {}) {
                // Elements
                this.form = document.getElementById('globalSearchForm');
                this.input = document.getElementById('globalSearchInput');
                this.button = document.getElementById('globalSearchButton');
                this.resultsContainer = document.getElementById('globalSearchResults');
                
                // Options with defaults
                this.options = {
                    apiEndpoint: options.apiEndpoint || '/api/products',
                    searchParam: options.searchParam || 'q',
                    minChars: options.minChars || 2,
                    debounceTime: options.debounceTime || 300,
                    resultLimit: options.resultLimit || 5,
                    searchResultsPage: options.searchResultsPage || '/products',
                    productPageUrl: options.productPageUrl || '/product',
                    searchInFields: options.searchInFields || ['name', 'description'],
                    placeholderImage: options.placeholderImage || 'https://via.placeholder.com/50',
                    onSelect: options.onSelect || this.defaultOnSelect.bind(this)
                };
                
                // State
                this.searchTimeoutId = null;
                this.lastQuery = '';
                this.searchResults = [];
                this.allProducts = []; // Will store all products once fetched
                this.isProductsLoaded = false;
                
                // Initialize
                this.init();
            }
            
            init() {
                // Form submit handler
                this.form.addEventListener('submit', this.handleFormSubmit.bind(this));
                
                // Input handlers
                this.input.addEventListener('input', this.handleInput.bind(this));
                this.input.addEventListener('focus', () => {
                    if (this.input.value.trim().length >= this.options.minChars) {
                        this.showResults();
                    }
                });
                
                // Close results when clicking outside
                document.addEventListener('click', (e) => {
                    if (!this.form.contains(e.target)) {
                        this.hideResults();
                    }
                });
                
                // Load products if not already available
                this.loadProducts();
            }
            
            async loadProducts() {
                // First, check if products are already loaded in the global scope
                if (window.allProducts && window.allProducts.length > 0) {
                    this.allProducts = window.allProducts;
                    this.isProductsLoaded = true;
                    return;
                }
                
                // If not, try to load from localStorage for faster subsequent loads
                const cachedProducts = localStorage.getItem('cachedProducts');
                if (cachedProducts) {
                    try {
                        this.allProducts = JSON.parse(cachedProducts);
                        this.isProductsLoaded = true;
                        // Still fetch in background to update cache
                        this.fetchProductsFromAPI(true);
                        return;
                    } catch (e) {
                        console.error('Error parsing cached products:', e);
                    }
                }
                
                // If no local cache, fetch from API
                await this.fetchProductsFromAPI();
            }
            
            async fetchProductsFromAPI(updateCacheOnly = false) {
                try {
                    const response = await fetch(this.options.apiEndpoint);
                    if (!response.ok) {
                        throw new Error('Failed to fetch products');
                    }
                    
                    const products = await response.json();
                    
                    // Add unique IDs if they don't exist (matching your existing code)
                    const processedProducts = products.map((product, index) => {
                        if (!product.id) {
                            product.id = index + 1;
                        }
                        return product;
                    });
                    
                    // Update cache
                    localStorage.setItem('cachedProducts', JSON.stringify(processedProducts));
                    
                    // Only update active products if not just refreshing cache
                    if (!updateCacheOnly) {
                        this.allProducts = processedProducts;
                        this.isProductsLoaded = true;
                        
                        // Also update global scope if it exists
                        if (typeof window.allProducts !== 'undefined') {
                            window.allProducts = processedProducts;
                        }
                    }
                } catch (error) {
                    console.error('Error fetching products:', error);
                    
                    // If we have no products data at all, show an error
                    if (!this.isProductsLoaded) {
                        console.warn('No product data available for search');
                    }
                }
            }
            
            handleFormSubmit(e) {
                e.preventDefault();
                const query = this.input.value.trim();
                
                if (query.length >= this.options.minChars) {
                    this.navigateToSearchResults(query);
                }
            }
            
            handleInput(e) {
                const query = this.input.value.trim();
                
                // Clear previous timeout
                if (this.searchTimeoutId) {
                    clearTimeout(this.searchTimeoutId);
                }
                
                if (query.length >= this.options.minChars) {
                    // Debounce search to avoid excessive processing
                    this.searchTimeoutId = setTimeout(() => {
                        if (query !== this.lastQuery) {
                            this.lastQuery = query;
                            this.performSearch(query);
                        }
                    }, this.options.debounceTime);
                } else {
                    this.hideResults();
                }
            }
            
            performSearch(query) {
                // Ensure products are loaded
                if (!this.isProductsLoaded) {
                    this.renderError('Loading products...');
                    // Try to load products again
                    this.loadProducts().then(() => {
                        if (this.isProductsLoaded) {
                            this.performSearch(query);
                        }
                    });
                    return;
                }
                
                // Perform client-side search
                this.searchResults = this.searchProducts(query);
                this.renderResults(query);
            }
            
            searchProducts(query) {
                query = query.toLowerCase();
                
                return this.allProducts.filter(product => {
                    return this.options.searchInFields.some(field => {
                        const value = product[field];
                        return value && String(value).toLowerCase().includes(query);
                    });
                });
            }
            
            renderResults(query) {
                if (this.searchResults.length === 0) {
                    this.resultsContainer.innerHTML = `
                        <div class="global-search-no-results">No products found for "${query}"</div>
                    `;
                } else {
                    // Show a limited number of results
                    const limitedResults = this.searchResults.slice(0, this.options.resultLimit);
                    const hasMoreResults = this.searchResults.length > this.options.resultLimit;
                    
                    let html = limitedResults.map((product, index) => `
                        <div class="global-search-result-item" data-id="${product.id}" data-index="${index}">
                            <img src="${product.imageUrl || this.options.placeholderImage}" 
                                alt="${product.name}" 
                                class="global-search-result-image"
                                onerror="this.src='${this.options.placeholderImage}'">
                            <div class="global-search-result-content">
                                <div class="global-search-result-title">${product.name}</div>
                                ${product.description ? 
                                    `<div class="global-search-result-description">${product.description}</div>` : ''}
                                <div class="global-search-result-price">₹ ${product.price}</div>
                            </div>
                        </div>
                    `).join('');
                    
                    // Add "See all results" link if there are more
                    if (hasMoreResults) {
                        html += `
                            <div class="global-search-all-results" id="seeAllResults">
                                See all ${this.searchResults.length} results for "${query}"
                            </div>
                        `;
                    }
                    
                    this.resultsContainer.innerHTML = html;
                    
                    // Add click handlers to results
                    const resultItems = this.resultsContainer.querySelectorAll('.global-search-result-item');
                    resultItems.forEach(item => {
                        item.addEventListener('click', () => {
                            const productId = item.dataset.id;
                            const index = parseInt(item.dataset.index);
                            this.options.onSelect(this.searchResults[index]);
                        });
                    });
                    
                    // Add click handler to "See all results"
                    const seeAllButton = document.getElementById('seeAllResults');
                    if (seeAllButton) {
                        seeAllButton.addEventListener('click', () => {
                            this.navigateToSearchResults(query);
                        });
                    }
                }
                
                this.showResults();
            }
            
            renderError(message) {
                this.resultsContainer.innerHTML = `
                    <div class="global-search-no-results">${message}</div>
                `;
                this.showResults();
            }
            
            showResults() {
                this.resultsContainer.classList.add('show');
            }
            
            hideResults() {
                this.resultsContainer.classList.remove('show');
            }
            
            navigateToSearchResults(query) {
                // Construct URL based on your existing implementation
                const url = `${this.options.searchResultsPage}?${this.options.searchParam}=${encodeURIComponent(query)}`;
                window.location.href = url;
            }
            
            defaultOnSelect(product) {
                // Navigate to the product page
                window.location.href = `${this.options.productPageUrl}/${product.id}`;
            }
            
            // Public method to update options
            updateOptions(newOptions) {
                this.options = {...this.options, ...newOptions};
            }
            
            // Public method to manually set product data
            setProductData(products) {
                this.allProducts = products;
                this.isProductsLoaded = true;
            }
        }

        // Initialize the global search component when the DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            // Create the global search instance
            window.globalSearch = new GlobalSearch({
                // Configuration settings - adjust these based on your actual endpoints and URLs
                apiEndpoint: '/api/products',
                searchResultsPage: '/products',
                productPageUrl: '/product',
                minChars: 2,
                resultLimit: 5,
                placeholderImage: 'https://via.placeholder.com/50',
                searchInFields: ['name', 'description', 'age']
            });
            
            // If products are already available in global scope, use them
            if (window.allProducts && window.allProducts.length > 0) {
                window.globalSearch.setProductData(window.allProducts);
            }
        });
    </script>
