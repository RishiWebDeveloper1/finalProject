    <script>
        // Global variables
        let totalAmount = 0;
        
        // Load cart from localStorage
        function loadCart() {
            // Get cart from localStorage
            const cartData = JSON.parse(localStorage.getItem('cart') || '[]');
            const cartContainer = document.getElementById("cart-container");
            const checkoutBtn = document.getElementById("checkout-btn");
            const cartSummary = document.getElementById("cart-summary");
            const continueShopping = document.getElementById("continue-shopping");
            
            cartContainer.innerHTML = "";
            
            if (cartData.length === 0) {
                cartContainer.innerHTML = `
                    <div class="empty-cart-message">
                        <i class="fas fa-shopping-cart"></i>
                        <p>Your cart is empty!</p>
                    </div>
                `;
                checkoutBtn.style.display = "none"; // Hide checkout button
                cartSummary.style.display = "none"; // Hide summary
                continueShopping.style.display = "block"; // Show continue shopping
                return;
            }
            
            checkoutBtn.style.display = "block"; // Show checkout button
            cartSummary.style.display = "block"; // Show summary
            continueShopping.style.display = "block"; // Show continue shopping
            
            let subtotal = 0;
            
            // Display each cart item
            cartData.forEach(item => {
                const quantity = item.quantity || 1;
                const itemTotal = item.price * quantity;
                subtotal += itemTotal;
                
                const cartItem = document.createElement("div");
                cartItem.classList.add("cart-item");
                
                // Use item properties as stored in localStorage
                cartItem.innerHTML = `
                    <img class="cart-item-image" src="${item.imageUrl || '/api/placeholder/100/100'}" alt="${item.name}">
                    <div class="cart-item-details">
                        <h3 class="cart-item-name">${item.name}</h3>
                        <p class="cart-item-price">₹${item.price.toLocaleString()}</p>
                        <div class="cart-item-quantity">
                            <span class="quantity-label">Quantity:</span>
                            <div class="quantity-controls">
                                <button class="quantity-btn" onclick="updateQuantity(${item.id}, -1)">
                                    <i class="fas fa-minus"></i>
                                </button>
                                <span class="quantity-value">${quantity}</span>
                                <button class="quantity-btn" onclick="updateQuantity(${item.id}, 1)">
                                    <i class="fas fa-plus"></i>
                                </button>
                            </div>
                            <button class="remove-btn" onclick="removeFromCart(${item.id})">
                                <i class="fas fa-trash-alt"></i> Remove
                            </button>
                        </div>
                    </div>
                `;
                
                cartContainer.appendChild(cartItem);
            });
            
            // Update the summary
            totalAmount = subtotal;
            document.getElementById("subtotal").textContent = `₹${subtotal.toLocaleString()}`;
            document.getElementById("total").textContent = `₹${subtotal.toLocaleString()}`;
        }
        
        // Update quantity
        function updateQuantity(productId, change) {
            // Get current cart
            let cart = JSON.parse(localStorage.getItem('cart') || '[]');
            
            // Find the item in the cart
            const itemIndex = cart.findIndex(item => item.id === productId);
            
            if (itemIndex !== -1) {
                // If item doesn't have a quantity property yet, add it
                if (!cart[itemIndex].hasOwnProperty('quantity')) {
                    cart[itemIndex].quantity = 1;
                }
                
                // Update quantity
                cart[itemIndex].quantity += change;
                
                // Remove item if quantity is 0 or less
                if (cart[itemIndex].quantity <= 0) {
                    cart.splice(itemIndex, 1);
                }
                
                // Save updated cart
                localStorage.setItem('cart', JSON.stringify(cart));
                
                // Update cart display
                loadCart();
                
                // Update cart counter in navbar (if exists on this page)
                updateCartCounter();
            }
        }
        
        // Update cart counter (in case we're showing it in the navbar)
        function updateCartCounter() {
            const cartCounter = document.getElementById('cart-nav-count');
            if (cartCounter) {
                const cart = JSON.parse(localStorage.getItem('cart') || '[]');
                
                if (cart.length > 0) {
                    cartCounter.textContent = cart.length;
                    cartCounter.style.display = 'flex';
                } else {
                    cartCounter.style.display = 'none';
                }
            }
        }
        
        // Remove item from cart
        function removeFromCart(productId) {
            // Get current cart
            let cart = JSON.parse(localStorage.getItem('cart') || '[]');
            
            // Remove the item
            cart = cart.filter(item => item.id !== productId);
            
            // Save updated cart
            localStorage.setItem('cart', JSON.stringify(cart));
            
            // Update cart display
            loadCart();
            
            // Update cart counter in navbar (if exists on this page)
            updateCartCounter();
        }
        
        // Start checkout process
        function startCheckout() {
            document.getElementById('cart-container').style.display = 'none';
            document.getElementById('cart-summary').style.display = 'none';
            document.getElementById('checkout-btn').style.display = 'none';
            document.getElementById('continue-shopping').style.display = 'none';
            document.getElementById('customer-info').style.display = 'block';
        }
        
        // Cancel checkout
        function cancelCheckout() {
            document.getElementById('cart-container').style.display = 'block';
            document.getElementById('cart-summary').style.display = 'block';
            document.getElementById('checkout-btn').style.display = 'block';
            document.getElementById('continue-shopping').style.display = 'block';
            document.getElementById('customer-info').style.display = 'none';
        }
        
        // Proceed to payment
        function proceedToPayment() {
            // Get customer details
            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const phone = document.getElementById('phone').value;
            const address = document.getElementById('address').value;
            
            // Validate form
            if (!name || !email || !phone || !address) {
                alert('Please fill in all fields');
                return;
            }
            
            // Initialize Razorpay payment
            initializeRazorpay(name, email, phone);
        }
        
        // Initialize Razorpay
        function initializeRazorpay(name, email, phone) {
            // Convert amount to paise (Razorpay requires amount in smallest currency unit)
            const amountInPaise = Math.round(totalAmount * 100);
            
            // You would normally get this key from your server for security
            const razorpayKey = 'YOUR_RAZORPAY_KEY_ID';
            
            // Create a new Razorpay instance
            const razorpay = new Razorpay({
                key: razorpayKey,
                amount: amountInPaise,
                currency: 'INR',
                name: 'Your Shop Name',
                description: 'Purchase',
                image: 'your_logo_url.png', // Your shop logo
                handler: function(response) {
                    // Payment success
                    handlePaymentSuccess(response);
                },
                prefill: {
                    name: name,
                    email: email,
                    contact: phone
                },
                theme: {
                    color: '#3498db'
                }
            });
            
            // Note: In a real implementation, you should create order on your server
            // And pass order_id to Razorpay
            
            razorpay.open();
        }
        
        // Handle payment success
        function handlePaymentSuccess(response) {
            // In a real app, you would verify payment with your server here
            
            // Generate a random order ID for this example
            const orderId = 'ORD' + Math.floor(Math.random() * 1000000);
            
            // Display success message
            document.getElementById('order-id').textContent = orderId;
            document.getElementById('payment-success').style.display = 'block';
            document.getElementById('customer-info').style.display = 'none';
            
            // Clear cart
            localStorage.setItem('cart', JSON.stringify([]));
            
            // Update cart counter
            updateCartCounter();
        }
        
        // Load cart when page opens
        document.addEventListener('DOMContentLoaded', function() {
            loadCart();
        });
        
        // Load cart immediately in case DOM is already loaded
        loadCart();
    </script>
